# Dynamic <span class="text-gradient font-normal">OpenGraph</span> images

```ts
// src/pages/blog/[id]/og.png
import { fontData } from "astro:assets";
import type { APIRoute } from "astro";
import { ImageResponse } from "@takumi-rs/image-response";
import ogImage from "../og-image";

export const GET: APIRoute = async (context) => {
  return new ImageResponse(ogImage(context.params.id), {
    width: 1200,
    height: 630,
    format: "png",
    fonts: await Promise.all(
      fontData["--font-montserrat"].map(async (font) => ({
        name: "Montserrat",
        weight: Number(font.weight),
        style: font.style,
        data: await fetch(new URL(font.src[0]!.url, context.url.origin)).then(
          (res) => res.arrayBuffer(),
        ),
      })),
    ),
  });
};
```
