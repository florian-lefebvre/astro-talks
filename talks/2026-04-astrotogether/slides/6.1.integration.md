# UI kit as an <span class="text-gradient font-normal">integration</span>

```ts
import { fontProviders } from "astro";

"astro:config:setup": (params) => {
  params.updateConfig({
    fonts: [
      {
        name: "Poppins",
        provider: fontProviders.local(),
        cssVariable: "--ui-kit-poppins",
        options: {
          variants: [
            {
              weight: "600",
              style: "normal",
              src: [new URL("./assets/fonts/poppins-600-normal.woff2", import.meta.url)],
            },
          ],
        },
      },
    ],
  });
}
```
