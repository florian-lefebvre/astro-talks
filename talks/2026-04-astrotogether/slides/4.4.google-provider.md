# Example: <span class="text-gradient font-normal">google provider</span>

Resolve options and fetch CSS

```ts
for (const format of options.formats) {
  const userAgent = userAgents[format];

  const rawCss = await $fetch<string>("/css2", {
    baseURL: "https://fonts.googleapis.com",
    headers: {
      "user-agent": userAgent,
    },
    query: {
      family: `${family}:${resolvedAxes.join(",")}@${resolvedVariants.join(";")}`,
      ...(glyphs && { text: glyphs }),
    },
  });
  // ...
}
```

Example URL:

```
https://fonts.googleapis.com/css2?family=Roboto:ital,wght@0,100..900;1,100..900&display=swap
```
