# Generating <span class="text-gradient font-normal">fallbacks</span>

```ts
import { fromBuffer } from "@capsizecss/unpack";

const ARIAL_METRICS = {
  ascent: 1854,
  descent: -434,
  lineGap: 67,
  unitsPerEm: 2048,
  xWidthAvg: 913,
};

if (fallback === "sans-serif") {
  const fallbackMetrics = await fromBuffer(fontBuffer);
  const cssProperties = computeCssProperties(ARIAL_METRICS, fallbackMetrics);
}
```
